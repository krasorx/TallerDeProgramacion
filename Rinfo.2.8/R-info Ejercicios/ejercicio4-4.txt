  proceso JuntarFlores(ES num:numero)

  fin
  proceso TirarFlores(E num:numero)

    repetir num
      depositarFlor
  fin
  proceso Rectangulo(ES alto:numero ; ES ancho : numero ; ES num:numero; ES avAnt:numero;ES caAnt:numero)
  variables
    caSig : numero
    avSig : numero
  comenzar
    repetir alto
      JuntarFlores(num)
      caSig := PosCa + 1
      BloquearEsquina(caSig,avSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir ancho
      avSig := PosAv + 1
      BloquearEsquina(caSig,avSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir alto
      JuntarFlores(num)
      caSig := PosCa - 1
      BloquearEsquina(caSig,avSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir ancho
      avSig := PosCa - 1
      BloquearEsquina(caSig,avSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
  fin

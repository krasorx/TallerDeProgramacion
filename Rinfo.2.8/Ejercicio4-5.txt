procesos
  proceso JuntarFlores(ES num:numero)
  comenzar
    mientras(HayFlorEnLaEsquina)
      tomarFlor
      num := num + 1
  fin
  proceso TirarFlores(ES num:numero)
  comenzar
    repetir num
      depositarFlor
      num := num - 1
  fin
  proceso Rectangulo(ES alto:numero ; ES ancho : numero ; ES num:numero; ES avAnt:numero;ES caAnt:numero)
  variables
    caSig : numero
    avSig : numero
  comenzar
    repetir alto
      JuntarFlores(num)
      caSig := PosCa + 1
      BloquearEsquina(PosAv,caSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir ancho
      avSig := PosAv + 1
      BloquearEsquina(avSig,PosCa)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir alto
      JuntarFlores(num)
      caSig := PosCa - 1
      BloquearEsquina(PosAv,caSig)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
    repetir ancho
      avSig := PosAv - 1
      BloquearEsquina(avSig,PosCa)
      mover
      LiberarEsquina(avAnt,caAnt)
      avAnt := PosAv
      caAnt := PosCa
    derecha
  fin
